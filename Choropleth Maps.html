#Get the data
data_url = "https://raw.githubusercontent.com/CHENG-WEI-HSUEH/Kaggle-Projects/main/World%20University%20Rankings/Data/school_and_country_table_1.csv"
distribution_schools = pd.read_csv(data_url,encoding="utf-8")
geo_json_url = "https://raw.githubusercontent.com/johan/world.geo.json/master/countries.geo.json"

'''
Sum the schools in same countries
'''
#Get countries list
Countries_list = pd.DataFrame(distribution_schools['country'].value_counts())
Countries_list.reset_index(level=0, inplace=True)
Countries_list.columns = ['Country', 'Number']

#show the number of uni by countries
pd.set_option("display.max_rows", None)
print(Countries_list)

#Turn into 3 digit codes (ISO 3166-1) 
countries = {}
for country in pycountry.countries:
    countries[country.name] = country.alpha_3

Countries_list['Country'] = [countries.get(country, 'Unknown code') for country in Countries_list['Country']]

#Testing code in order to find invalid name -- index: [ 0 10 11 20 21 22 65]
#print(f"Unknown index: {Countries_list[Countries_list['Country'] == 'Unknown code'].index.values}")
Countries_list.loc[0,'Country'] = 'USA'
Countries_list.loc[10,'Country'] = 'TWN'
Countries_list.loc[11,'Country'] = 'KOR'
Countries_list.loc[20,'Country'] = 'IRL'
Countries_list.loc[21,'Country'] = 'CZE'
Countries_list.loc[22,'Country'] = 'IRN'
Countries_list.loc[65,'Country'] = 'MAC'

#Set the center of the map
World_map = folium.Map(location=[52.52, 13.40], zoom_start=3)

folium.Choropleth(
    geo_data=geo_json_url,
    name="choropleth",
    data=Countries_list,
    columns=["Country", "Number"],
    key_on="feature.id",
    fill_color="YlGn",
    nan_fill_color="white",
    fill_opacity=0.5,
    line_opacity=0,
    legend_name="The number of schools",
).add_to(World_map)

folium.LayerControl().add_to(World_map)
